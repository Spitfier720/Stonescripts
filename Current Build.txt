import Cosmetics/Hats/Helmet

?hp <= 10
    activate potion

?foe.distance > 11
    equipL crossbow
    equipR dashing shield
:
    loadout 3

?pickup.distance < 15
    equipL star stone

?item.right.state = 3
    equip quarterstaff
    equip @item.right@

?item.GetCooldown("blade") <= 0
^ & foe.count > 7
    equip blade
    activate R

?item.GetCooldown("skeleton_arm") <= 0
^ & foe.distance <= 6
    equip skeleton_arm

    ?item.CanActivate("skeleton_arm")
        activate R

?foe = boss & item.GetCooldown("bardiche") <= 0
^ & foe.distance <= 9
    equip bardiche

    ?item.CanActivate("bardiche")
        activate R
