import Cosmetics/Hats/Helmet
import Cosmetics/Hats/Skully
import Cosmetics/PetBoo

var HEALTH_THRESHOLD = 5

func usePotion()
    ?hp <= HEALTH_THRESHOLD
        activate potion

func baseLoadout()
    ?foe.distance > 11
        equipL crossbow
        equipR dashing shield
    :
        loadout 3

    ?pickup.distance < 15
        equipL star stone

func aacTwoHand()
    ?item.right.state = 3
        equip quarterstaff
        equip @item.right@

func activateBlade()
    ?item.GetCooldown("blade") <= 0 & foe.count > 7
        var slideCount = 0
        ?slideCount = 0
            screen.Next()
            slideCount = 20
        
        ?slideCount > 0
            slideCount--
            
            ?slideCount = 0 & foe.count >= 15 // foe count can be removed for this part
                equip blade
                activate R
            screen.ResetOffset()

// ?item.GetCooldown("skeleton_arm") <= 0
// ^ & foe.distance <= 6
//     equip skeleton arm

//     ?item.CanActivate("skeleton_arm")
//         activate R

func activateBardiche()
    > @item.GetCooldown("bardiche")@
    ?foe = boss & item.GetCooldown("bardiche") <= 0
    ^ & foe.distance <= 9
        equip bardiche
        activate R

func activateCultist()
    > @item.GetCooldown("cultist")@
    ?foe = boss & item.GetCooldown("cultist") <= 0
    ^ & foe.distance < 15
        equip cultist
        activate R

func main()
    usePotion()
    baseLoadout()
    aacTwoHand() //aac for left and right hand to be implemented
    activateBlade()
    // activateSkeletonArm() // maybe we implement this later
    activateBardiche()
    activateCultist()

main()
